async function getRecetteToDay() {
    try {
        const [recetteRes, depenseRes] = await Promise.all([
            fetch('../../API/admin/detailFactureToDay.php'),
            fetch('../../API/admin/depenseToDay.php')
        ]);

        const recettes = await recetteRes.json();
        const depenses = await depenseRes.json();

        const tbody = document.getElementById('recetteBody');
        tbody.innerHTML = '';

        let totalEntreeUSD = 0;
        let totalEntreeCDF = 0;
        let totalSortieUSD = 0;
        let totalSortieCDF = 0;

        // --- Recettes (Entrées) ---
        recettes.forEach(item => {
            const row = document.createElement('tr');
            row.classList.add('hover:bg-green-50');

            row.innerHTML = `
                <td class="px-6 py-3 font-medium">${item.libelle}</td>
                <td class="px-6 py-3 text-right text-blue-700 font-semibold">${item.total_usd}</td>
                <td class="px-6 py-3 text-right text-emerald-700 font-semibold">${item.total_cdf}</td>
                <td class="px-6 py-3 text-right">-</td>
                <td class="px-6 py-3 text-right">-</td>
            `;
            tbody.appendChild(row);

            totalEntreeUSD += parseFloat(item.total_usd) || 0;
            totalEntreeCDF += parseFloat(item.total_cdf) || 0;
        });

        // --- Dépenses (Sorties) ---
        depenses.forEach(item => {
            const row = document.createElement('tr');
            row.classList.add('hover:bg-red-50');

            row.innerHTML = `
                <td class="px-6 py-3 font-medium">${item.libelle}</td>
                <td class="px-6 py-3 text-right">-</td>
                <td class="px-6 py-3 text-right">-</td>
                <td class="px-6 py-3 text-right text-blue-700 font-semibold">${item.total_usd}</td>
                <td class="px-6 py-3 text-right text-emerald-700 font-semibold">${item.total_cdf}</td>
            `;
            tbody.appendChild(row);

            totalSortieUSD += parseFloat(item.total_usd) || 0;
            totalSortieCDF += parseFloat(item.total_cdf) || 0;
        });

        // --- Aucun résultat ---
        if (recettes.length === 0 && depenses.length === 0) {
            const row = document.createElement('tr');
            row.innerHTML = `<td class="px-6 py-4 text-center" colspan="5">Aucune opération pour aujourd'hui</td>`;
            tbody.appendChild(row);
        }

        // --- Footer avec totaux ---
        const tfoot = document.getElementById('journalFooter');
        if(tfoot){
            tfoot.innerHTML = `
                <tr class="bg-gray-100 font-semibold">
                    <td class="px-6 py-3 text-right">TOTAL</td>
                    <td class="px-6 py-3 text-right text-blue-700">${totalEntreeUSD.toLocaleString('fr-FR', {minimumFractionDigits:2})} $</td>
                    <td class="px-6 py-3 text-right text-emerald-700">${totalEntreeCDF.toLocaleString('fr-FR', {minimumFractionDigits:2})} FC</td>
                    <td class="px-6 py-3 text-right text-blue-700">${totalSortieUSD.toLocaleString('fr-FR', {minimumFractionDigits:2})} $</td>
                    <td class="px-6 py-3 text-right text-emerald-700">${totalSortieCDF.toLocaleString('fr-FR', {minimumFractionDigits:2})} FC</td>
                </tr>
                <tr class="bg-gray-200 font-bold">
                    <td class="px-6 py-3 text-right" colspan="2">SOLDE</td>
                    <td class="px-6 py-3 text-right text-blue-700">${(totalEntreeUSD - totalSortieUSD).toLocaleString('fr-FR', {minimumFractionDigits:2})} $</td>
                    <td class="px-6 py-3 text-right text-emerald-700" colspan="2">${(totalEntreeCDF - totalSortieCDF).toLocaleString('fr-FR', {minimumFractionDigits:2})} FC</td>

    }

async function MedicamentMoinsDemander(){
    try {
        const res = await fetch("../../API/pharmacie/medicamentMoinsDemade.php");
        const data = await res.json();
        const tableMediMoinsDemander = document.getElementById('MedicamentMoinsDemander');
        tableMediMoinsDemander.innerHTML = "";


        if(data.length === 0){
            const noData = document.createElement('tr');
            noData.innerHTML = `<td class="px-6 py-4 whitespace-nowrap text-sm text-center-500" colspan="6">Aucun Medicament trouvée</td>`;
            tableMediMoinsDemander.appendChild(noData);
        }else{
            data.forEach(medicament =>{
                const row = document.createElement('tr');
                row.classList.add('hover:bg-gray-100');

                row.innerHTML = `<td class="px-6 py-4 whitespace-nowrap text-sm font-medium text-gray-900">${medicament.nom_medicament}  ${medicament.dosage}</td>
                                <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500">${medicament.forme} </td>
                                <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-900">${medicament.total_sortie}</td>`
                tableMediMoinsDemander.appendChild(row);
            })
         
        }
        
    } catch (error) {
        console.error("Erreur fetch:", error);
    }
}

let soldeHebdoChart = null;
async function afficherSoldeHebdo() {
    try {
        const res = await fetch("../../API/admin/solde_semaine.php");
        const data = await res.json();

        const labels = data.map(item => item.jour);
        const valeurs = data.map(item => item.solde);

        // Détruire l'ancien graphique si existe
        if (soldeHebdoChart !== null) {
            soldeHebdoChart.destroy();
        }

        const ctx = document.getElementById('soldeHebdoChart').getContext('2d');
        soldeHebdoChart = new Chart(ctx, {
            type: 'bar', // tu peux mettre 'line' pour un graphique en courbes
            data: {
                labels: labels,
                datasets: [{
                    label: 'Solde par jour (Fc)',
                    data: valeurs,
                    backgroundColor:
                     [
                        "rgba(153, 102, 255, 0.2)",
                        "rgba(201, 203, 207, 0.2)",
                        "rgba(255, 99, 132, 0.2)",
                        "rgba(255, 159, 64, 0.2)",
                        "rgba(255, 205, 86, 0.2)",
                        "rgba(75, 192, 192, 0.2)",
                        "rgba(121, 85, 72, 0.2)"
                       
                    ],
                    borderColor: [
                        
                        "rgba(153, 102, 255,1)",
                        "rgba(201, 203, 207,1)",
                        "rgba(255, 99, 132,1)",
                        "rgba(255, 205, 86,1)",
                         "rgba(255, 159, 64,1)",
                        "rgba(75, 192, 192,1)",
                        "rgba(121, 85, 72,1)"
                    ],
                    borderWidth: 1
                }]
            },
            options: {
                responsive: true,
                plugins: {
                    legend: {
                        display: true
                    }
                },
                scales: {
                    y: {
                        beginAtZero: true
                    }
                }
            }
        });

    } catch (error) {
        console.error("Erreur fetch :", error);
    }
}
let  sortieHebdoChart = null;
async function afficherSortieHebdo() {
    try {
        const res = await fetch("../../API/admin/sortie.semaine.php");
        const data = await res.json();

        const labels = data.map(item => item.jour);
        const valeurs = data.map(item => item.total_sortie);

        const couleurs = {
            "Lundi": "rgba(201, 203, 207, 0.2)",
            "Mardi": "rgba(255, 159, 64, 0.2)",
            "Mercredi": "rgba(255, 205, 86, 0.2)",
            "Jeudi": "rgba(75, 192, 192, 0.2)",
            "Vendredi": "rgba(153, 102, 255, 0.2)",
            "Samedi": "rgba(121, 85, 72, 0.2)",
            "Dimanche": "rgba(255, 99, 132, 0.2) "
        };

        const couleursBorder = {
            "Lundi": "rgba(201, 203, 207)",
            "Mardi": "rgba(255, 159, 64)",
            "Mercredi": "rgba(255, 205, 86)",
            "Jeudi": "rgba(75, 192, 192)",
            "Vendredi": "rgba(153, 102, 255)",
            "Samedi": "rgba(121, 85, 72)",
            "Dimanche": " rgba(255, 99, 132)"
        };

        const backgroundColors = labels.map(j => couleurs[j]);
        const borderColors = labels.map(j => couleursBorder[j]);

        const ctx = document.getElementById('sortieHebdoChart').getContext('2d');

        if (sortieHebdoChart) {
            // âÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂ Mettre à jour les données existantes
            sortieHebdoChart.data.labels = labels;
            sortieHebdoChart.data.datasets[0].data = valeurs;
            sortieHebdoChart.data.datasets[0].backgroundColor = backgroundColors;
            sortieHebdoChart.data.datasets[0].borderColor = borderColors;
            sortieHebdoChart.update(); // Rafraîchissement du graphique
        } else {
            // âÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂÂ Création initiale du graphique
            sortieHebdoChart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: labels,
                    datasets: [{
                        label: 'Sorite de la semaine',
                        data: valeurs,
                        backgroundColor: backgroundColors,
                        borderColor: borderColors,
                        borderWidth: 1
                    }]
                },
                options: {
                    responsive: true,
                    plugins: {
                        legend: { display: true },
                        tooltip: { enabled: true }
                    },
                    scales: {
                        y: { beginAtZero: true }
                    }
                }
            });
        }

    } catch (error) {
        console.error("Erreur fetch :", error);
    }
}

    function openChart(chartContainerId) {
        const chartContainers = document.querySelectorAll('.chart-container');
        chartContainers.forEach(container => {
            container.classList.remove('active');
            if (container.id === chartContainerId) {
                container.classList.add('active');
            }
        });
    }

    function openSection(event, sectionId){
        const sectionContainer = document.querySelectorAll(".sectionContent");
        sectionContainer.forEach(section =>{
            section.classList.remove('active');
            if(section.id === sectionId ){
                section.classList.add("active");
            }
        })
          const sectionBtn = document.querySelectorAll(".sectionBtn");
          sectionBtn.forEach(btn =>{
            btn.classList.remove('text-blue-600','border-blue-600','border-b-2');
            btn.classList.add('text-gray-600','hover:text-blue-600');
          });

          event.currentTarget.classList.remove('text-gray-600','hover:text-blue-600');
          event.currentTarget.classList.add('text-blue-600','border-blue-600','border-b-2');
    }

    async function MouvementDuJours() {

        const sorties = data.map(item => item.total_sortie);

        // Création du graphique
        const ctx = document.getElementById("mouvementJours").getContext("2d");

          type: "line", // tu peux mettre "bar" si tu veux un histogramme
          data: {
            labels: labels,
            datasets: [
              {
                label: "Entrées",
                data: entrees,

                fill: true,
                tension: 0.3
              },
              {
